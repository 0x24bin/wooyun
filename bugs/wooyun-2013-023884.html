后台登录效验太简单了，

$("#signupsubmit").click(

	function(){

		userLogin();

	}); 

	function userLogin(){

		if(!validator.form()) return;

		parm=$("#signupform").serialize();

		

		$.ajax({

		   type: "Get",

		   url: "LoginJson.action",

		   data: parm, 

		   success: function(msg){

			   if (msg==1){

				  $("#signupform").submit();

		   		}

			   else

		   		  alert( msg );

		   },

		   error:function(){

			   alert("网络连接超时!");

		   }

		});

	}
