function numTable($table='',$wheres=false)

	{

		$table = $this->dbPrefix.$table;

		$sql = "SELECT COUNT(*) AS num FROM `$table`";

		if($wheres)

		{

			$sql .= " WHERE ";

			if(is_array($wheres))

			{

				foreach($wheres as $key => $val)

				{

					$whr[] = "`$key`='".$val."'";

				}

				$sql .= implode(" AND ",$whr);

			}

			elseif(is_string($wheres))

			{

				$sql .= $wheres;

			}

		}

	}
