function setconfig($string) {

  if(!get_magic_quotes_gpc()) {

    $string = str_replace('\'', '\\\'', $string);

  } else {

    $string = str_replace('\"', '"', $string);

  }

  return $string;

}



function astrfilter($str) {

  if(is_array($str)) {

    foreach($str as $key => $val) {

      $str[$key] = astrfilter($val);

    }

  } else {

    $str = str_replace(Array('eval'),'',$str);    

  }

  return $str;

}
