function getRealFields($src_fields_list)

    {

        $fields = $src_fields_list;

        if (!$src_fields_list)

        {

            $fields = '';

        }

        $fields = preg_replace('/([a-zA-Z0-9_]+)\.([a-zA-Z0-9_*]+)/e', "\$this->_getFieldTable('\\1') . '.\\2'", $fields);//对注射语句没有影响



        return $fields;

}



function _getFieldTable($owner)
