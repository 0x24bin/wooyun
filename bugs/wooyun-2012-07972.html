<script type="text/javascript" src="http://s2.pinglun.la/md/pinglun.la.js" charset="utf-8"></script>

$webpage_url = "http://".PLL_URL."/comment_box/plugin/get_comments?url=".urlencode(pinglunla_cur_page_url());



?>
