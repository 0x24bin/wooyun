漏洞证明：

#include <stdlib.h>

#include <dirent.h>

#include <sys/stat.h>

#include <string.h>



int main()

{

    int fsize = 0;

    FILE *fp = fopen("文件路径", "r");

    if (fp)

    {

        fseek(fp, 0, SEEK_END);  

        fsize = ftell(fp); 

        //printf("文件%s可读  长度:%d\n", fp ? "" : "不", fsize);

        char *buf = (char*)malloc(fsize);

        fseek(fp, 0, 0); 

        fread(buf, 1, fsize, fp);

        for (int i = 0;i < fsize;i++)

        {

            if (i % 0x10 == 0)

            {

                //printf("\n");

            }

            printf("\\x%02X", buf[i] & 0xFF);

        }

    }

    return 0;

}
