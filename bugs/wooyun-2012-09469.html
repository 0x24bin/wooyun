function kc_stripslashes_array(&$_data){

	if (is_array($_data)){

		foreach ($_data as $_key => $_value){

			$_data[$_key]=kc_stripslashes_array($_value);

		}

		return $_data;

	}else{

		return stripslashes($_data);

	}

}
